  <script>
    // Simple timeline using Web Animations API
    const title = document.querySelector('h1');
    const subtitle = document.querySelector('.subtitle');
    const closing = document.querySelector('.closing');
    const bar = document.getElementById('bar');
    const stage = document.getElementById('stage');

    function animateSparkles(){
      const emojis = ['âœ¨','ðŸ’«','ðŸŒŸ','ðŸ’–'];
      for(let i=0;i<10;i++){
        const s = document.createElement('div');
        s.className = 'sparkle';
        s.textContent = emojis[Math.floor(Math.random()*emojis.length)];
        s.style.left = Math.random()*90+5 + '%';
        s.style.top = Math.random()*70+10 + '%';
        stage.appendChild(s);
        s.animate([{opacity:0, transform:'translateY(6px) scale(.8)'},{opacity:1, transform:'translateY(-8px) scale(1)'},{opacity:0, transform:'translateY(-16px) scale(.9)'}],{duration:1200+Math.random()*600, delay:200*i/3, easing:'ease-out'}).finished.then(()=>s.remove());
      }
    }

    function timeline(){
      bar.style.width = '0%';
      // Title
      title.animate([{opacity:0, transform:'translateY(24px)'},{opacity:1, transform:'translateY(0)'}],{duration:900, easing:'cubic-bezier(.2,.7,.2,1)'});
      // progress
      setTimeout(()=> bar.style.width = '35%', 200);

      // Subtitle
      setTimeout(()=>{
        subtitle.animate([{opacity:0, transform:'translateY(18px)'},{opacity:1, transform:'translateY(0)'}],{duration:800, easing:'cubic-bezier(.2,.7,.2,1)'});
        animateSparkles();
      }, 800);
      setTimeout(()=> bar.style.width = '70%', 1000);

      // Closing
      setTimeout(()=>{
        closing.animate([{opacity:0},{opacity:1}],{duration:700, easing:'ease-out'});
        animateSparkles();
      }, 1700);
      setTimeout(()=> bar.style.width = '100%', 1800);
    }

    // Reset initial states so replay works cleanly
    function resetStates(){
      title.style.opacity = 0; title.style.transform = 'translateY(24px)';
      subtitle.style.opacity = 0; subtitle.style.transform = 'translateY(18px)';
      closing.style.opacity = 0; closing.style.transform = 'none';
      // After anims, apply end state via onfinish
      const end = (el)=> el.animate([{opacity:1, transform:'none'}],{duration:1, delay:1900, fill:'forwards'});
      end(title); end(subtitle); end(closing);
    }

    // Accessibility: reduce motion if user prefers
    const media = window.matchMedia('(prefers-reduced-motion: reduce)');

    function play(){
      if(media.matches){
        // minimal motion
        title.style.opacity = 1; subtitle.style.opacity = 1; closing.style.opacity = 1; bar.style.width = '100%';
      } else {
        resetStates(); timeline();
      }
    }

    // Buttons
    document.getElementById('replay').addEventListener('click', play);
    document.getElementById('share').addEventListener('click', async () => {
      try{
        const shareUrl = window.location.href;
        await navigator.clipboard.writeText(shareUrl);
        alert('Link copied! Share it with Monisa.');
      }catch(e){ alert('Copy failed. You can still share this page URL.'); }
    });

    // Autoâ€‘play on load
    window.addEventListener('load', play);
  </script>
